rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isOwner(teacherId) {
      return request.auth != null && request.auth.uid == teacherId;
    }

    /*
      Existing stable single-teacher app.
      Keep this so your current working version does not break.
    */
    match /students/{studentId} {
      allow read: if true;
      allow create, update, delete: if signedIn();
    }

    match /sections/{sectionId} {
      allow read: if true;
      allow create, update, delete: if signedIn();
    }

    match /recitationLogs/{logId} {
      allow read: if signedIn();
      allow create: if signedIn();
      allow update, delete: if false;
    }

    /*
      New multi-teacher app.
      Each teacher writes only to their own teacher UID path.
    */
    match /teachers/{teacherId} {
      allow read: if true;
      allow create, update: if isOwner(teacherId);
      allow delete: if false;

      match /students/{studentId} {
        allow read: if true;
        allow create, update, delete: if isOwner(teacherId);
      }

      match /sections/{sectionId} {
        allow read: if true;
        allow create, update, delete: if isOwner(teacherId);
      }

      match /recitationLogs/{logId} {
        allow read: if isOwner(teacherId);
        allow create: if isOwner(teacherId);
        allow update, delete: if false;
      }
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
